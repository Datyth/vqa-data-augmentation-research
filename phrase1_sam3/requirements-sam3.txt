# Notes for real SAM3 inference in this project.
#
# Use the existing datpt_rs_aug environment. It is Python 3.12, which matches
# the official SAM3 prerequisite. SAM3 also needs PyTorch >=2.7 and a CUDA GPU
# with CUDA >=12.6.
#
# Official installation flow used here:
#   conda activate datpt_rs_aug
#   pip install torch==2.10.0 torchvision --index-url https://download.pytorch.org/whl/cu128
#   git clone https://github.com/facebookresearch/sam3.git /tmp/facebookresearch-sam3
#   pip install /tmp/facebookresearch-sam3
#   pip install "setuptools<81" einops pycocotools psutil
#   pip install -r phrase1_sam3/requirements.txt
#   python -m spacy download en_core_web_sm
#   hf auth login
#
# Checkpoint access is gated. Request access to the official Hugging Face SAM3
# checkpoint repo before running the backend. The model builder downloads the
# checkpoint the first time `sam_backend=sam3` is used.
#
# Optional faster inference deps from the official README:
#   pip install einops ninja
#   pip install flash-attn-3 --no-deps --index-url https://download.pytorch.org/whl/cu128
#   pip install git+https://github.com/ronghanghu/cc_torch.git
